104void convert(LCDObj_t * inputs){
105 // split 1st and 2nd 4 bits light on/off enable low read/write register select
106 inputs->data[0] = ((0b11110000 & inputs->databus) | (inputs->light << 3) | (1 << 2) | (inputs->rnw << 1) | (inputs->reg_sel << 0));
107 inputs->data[1] = ((0b11110000 & inputs->databus) | (inputs->light << 3) | (inputs->rnw << 1) | (inputs->reg_sel << 0));
108 inputs->data[2] = ((0b11110000 & (inputs->databus << 4 )) | (inputs->light << 3) | (1 << 2) | (inputs->rnw << 1) | (inputs->reg_sel << 0));
109 inputs->data[3] = ((0b11110000 & (inputs->databus << 4 )) | (inputs->light << 3) | (inputs->rnw << 1) | (inputs->reg_sel << 0));
110 inputs->databus = 0x00;
111 inputs->rnw = 0;
112 inputs->reg_sel = 0;
113 inputs->enable = 0;
114}
